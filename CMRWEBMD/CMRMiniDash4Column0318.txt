# --- Win32 focus helpers ---
Add-Type @"
using System; using System.Runtime.InteropServices;
public static class Win32 {
  [DllImport("user32.dll")] public static extern IntPtr GetForegroundWindow();
  [DllImport("user32.dll")] public static extern bool SetForegroundWindow(IntPtr hWnd);
  [DllImport("user32.dll")] public static extern bool ShowWindowAsync(IntPtr hWnd, int nCmdShow);
}
"@
$script:lastHwnd=[IntPtr]::Zero
function Get-FG { [Win32]::GetForegroundWindow() }
function Focus-LastWindow {
  try {
    $fg=Get-FG
    if ($fg -ne [IntPtr]::Zero -and $script:form -and $fg -ne $script:form.Handle) { $script:lastHwnd=$fg }
    if ($script:lastHwnd -eq [IntPtr]::Zero) { return $false }
    if ($fg -ne $script:lastHwnd) { [void][Win32]::ShowWindowAsync($script:lastHwnd,5); [void][Win32]::SetForegroundWindow($script:lastHwnd) }
    return $true
  } catch { Write-Log "Focus-LastWindow: $($_.Exception.Message)"; return $false }
}
function Ensure-TargetFast {
  try {
    $fg=Get-FG
    if ($fg -eq [IntPtr]::Zero) { return Focus-LastWindow }
    if ($script:form -and $fg -eq $script:form.Handle) { return Focus-LastWindow }
    $script:lastHwnd=$fg; return $true
  } catch { return $false }
}

using System; using System.Runtime.InteropServices;
public static class Hotkeys {
  [DllImport("user32.dll")] public static extern void keybd_event(byte bVk, byte bScan, int dwFlags, int dwExtraInfo);
  public const byte VK_LWIN = 0x5B; public const byte VK_H = 0x48;
  public const int KEYEVENTF_EXTENDEDKEY=0x1; public const int KEYEVENTF_KEYUP=0x2;
  public static void WinH(){
    keybd_event(VK_LWIN,0,KEYEVENTF_EXTENDEDKEY,0);
    keybd_event(VK_H,0,KEYEVENTF_EXTENDEDKEY,0);
    keybd_event(VK_H,0,KEYEVENTF_KEYUP,0);
    keybd_event(VK_LWIN,0,KEYEVENTF_KEYUP,0);
  }
}
"@

function Get-FocusedSelectionText { if (-not (Ensure-UIA)) { return $null }
  try { $ae=[System.Windows.Automation.AutomationElement]::FocusedElement
    if ($null -eq $ae) { return $null }
    $tp=$null
    if ($ae.TryGetCurrentPattern([System.Windows.Automation.TextPattern]::Pattern,[ref]$tp)) { $sel=$tp.GetSelection(); if ($sel -and $sel.Length -gt 0) { return $sel[0].GetText(-1) } }
    $vp=$null
    if ($ae.TryGetCurrentPattern([System.Windows.Automation.ValuePattern]::Pattern,[ref]$vp)) { return $vp.Current.Value }
  } catch { Write-Log "Get-FocusedSelectionText: $($_.Exception.Message)" }
  $null }

# --- Safe buffer + copy/paste ---
$script:SafeBuffer=""
function SafeCopy-Selection {
  $uia=Get-FocusedSelectionText
  if ($uia -and $uia.Length -gt 0) { $script:SafeBuffer=$uia; try{ [System.Windows.Forms.Clipboard]::SetText($uia) }catch{}; return }
  if (-not (Focus-LastWindow)) { return }
  try {
    [System.Windows.Forms.SendKeys]::SendWait('^c'); Start-Sleep -Milliseconds 60
    $t=''; try{ if([System.Windows.Forms.Clipboard]::ContainsText()){ $t=[System.Windows.Forms.Clipboard]::GetText() } }catch{}
    $script:SafeBuffer=$t
  } catch {}
}
function SafePaste-Buffer { if ([string]::IsNullOrEmpty($script:SafeBuffer)) { return }
  if (-not (Ensure-TargetFast)) { return }
  try { [System.Windows.Forms.SendKeys]::Send($script:SafeBuffer) } catch {} }

# --- Send helpers ---
function SendKeys-ToLastWindow([string]$keys,[bool]$fast=$false){ if($fast){ if(-not (Ensure-TargetFast)){return} } else { if(-not (Focus-LastWindow)){return} }
  try { if($fast){[System.Windows.Forms.SendKeys]::Send($keys)} else {[System.Windows.Forms.SendKeys]::SendWait($keys)} } catch {} }
function TypeText-ToLastWindow([string]$text,[bool]$enter=$false,[bool]$fast=$false){ if($fast){ if(-not (Ensure-TargetFast)){return} } else { if(-not (Focus-LastWindow)){return} }
  try { if($fast){[System.Windows.Forms.SendKeys]::Send($text)} else {[System.Windows.Forms.SendKeys]::SendWait($text)}; if($enter){ if($fast){[System.Windows.Forms.SendKeys]::Send('{ENTER}')} else {[System.Windows.Forms.SendKeys]::SendWait('{ENTER}')} } } catch {} }

function Play-PayloadWithSleep([string]$payload) {
    if ([string]::IsNullOrWhiteSpace($payload)) { return }

    # Split the payload on tokens like {SLEEP 500} but keep the tokens
    $parts = [System.Text.RegularExpressions.Regex]::Split(
        $payload, '(\{SLEEP\s+\d+\})'
    ) | Where-Object { $_ -ne '' }

    foreach ($p in $parts) {
        if ($p -match '^\{SLEEP\s+(\d+)\}$') {
            $ms = [int]$Matches[1]
            Start-Sleep -Milliseconds $ms
        } else {
            # Send the fragment as SendKeys (supports {TAB}, {ENTER}, ^c, etc.)
            SendKeys-ToLastWindow $p $false
        }
    }
}

# --- Custom action executor ---
function Invoke-CustomAction([string]$payload,[string]$mode='auto'){
  if([string]::IsNullOrWhiteSpace($payload)){return}
  switch ($mode) { 'text' { TypeText-ToLastWindow $payload $false $false }
    'sendkeys' { Play-PayloadWithSleep $payload }
    default { if($payload -match '[\{\}\^\%\+]'){ SendKeys-ToLastWindow $payload $false } else { TypeText-ToLastWindow $payload $false $false } } } }

function Load-JsonArray([string]$path){ try { if(Test-Path $path){ $raw=Get-Content -Path $path -Raw; if($raw -and $raw.Trim().Length -gt 0){ $p=$raw | ConvertFrom-Json; if($p -is [System.Array]){ return ,$p } elseif($p){ return ,@($p) } } } } catch { Write-Log "Load-JsonArray: $($_.Exception.Message)" } @() }

# Helper: keep window fully visible on its current screen
function Clamp-FormToScreen([System.Windows.Forms.Form]$f,[System.Windows.Forms.Screen]$scr){
    try{
        $wa = $scr.WorkingArea
        $x = [Math]::Min([Math]::Max($f.Left, $wa.X), $wa.Right - $f.Width)
        $y = [Math]::Min([Math]::Max($f.Top,  $wa.Y), $wa.Bottom - $f.Height)
        $f.Location = [System.Drawing.Point]::new([int]$x,[int]$y)
    }catch{}
}

# Track current screen and scale proportionally when crossing to a different monitor
$script:CurrentScreen = [System.Windows.Forms.Screen]::FromControl($script:form)
$script:form.Add_Move({
    try{
        $scr = [System.Windows.Forms.Screen]::FromControl($script:form)
        if($script:CurrentScreen -and $scr.DeviceName -ne $script:CurrentScreen.DeviceName){
            $old = $script:CurrentScreen.WorkingArea
            $new = $scr.WorkingArea
            if($old.Width -gt 0 -and $old.Height -gt 0){
                $scaleX = [double]$new.Width / [double]$old.Width
                $scaleY = [double]$new.Height / [double]$old.Height
                $nw = [int][Math]::Max($script:form.MinimumSize.Width,  [Math]::Round($script:form.Width  * $scaleX))
                $nh = [int][Math]::Max($script:form.MinimumSize.Height, [Math]::Round($script:form.Height * $scaleY))
                # Allow temporary unlock to apply size
                $script:form.MinimumSize = [System.Drawing.Size]::Empty
                $script:form.MaximumSize = [System.Drawing.Size]::Empty
                $script:form.Size = [System.Drawing.Size]::new($nw,$nh)
                $script:form.MinimumSize = $script:BaseSize
                $script:form.MaximumSize = [System.Drawing.Size]::Empty
            }
            $script:CurrentScreen = $scr
            Clamp-FormToScreen $script:form $scr
        }
        else{
            # Same screen move: just ensure in-bounds
            Clamp-FormToScreen $script:form $scr
        }
    }catch{}
})

function Add-EmbeddedPair(

# Auto-fit the main form height to show all controls on a tab (up to a max client height)
function Fit-FormHeight-ToTab([System.Windows.Forms.Form]$form,[System.Windows.Forms.TabPage]$tp,[int]$maxClientH=980){
  try{
    $maxBottom = 0
    foreach($c in $tp.Controls){
      if(-not $c.Visible){ continue }
      if($c -is [System.Windows.Forms.Panel] -and $c.Name -eq 'BottomBar'){ continue }
      if($c.Bottom -gt $maxBottom){ $maxBottom = $c.Bottom }
    }
    $needClientH = [int]($maxBottom + 20)
    $needClientH = [Math]::Min($needClientH, $maxClientH)
    $delta = $form.Height - $form.ClientSize.Height
    if($needClientH -gt $form.ClientSize.Height){
      $form.MinimumSize = [System.Drawing.Size]::Empty
      $form.MaximumSize = [System.Drawing.Size]::Empty
      $form.Size = [System.Drawing.Size]::new($form.Width, $needClientH + $delta)
      $form.MinimumSize = $form.Size
      $form.MaximumSize = [System.Drawing.Size]::Empty
    }
  }catch{}
}
