# --- Win32 focus helpers ---
Add-Type @"
using System; using System.Runtime.InteropServices;
public static class Win32 {
  [DllImport("user32.dll")] public static extern IntPtr GetForegroundWindow();
  [DllImport("user32.dll")] public static extern bool SetForegroundWindow(IntPtr hWnd);
  [DllImport("user32.dll")] public static extern bool ShowWindowAsync(IntPtr hWnd, int nCmdShow);
}
"@
$script:lastHwnd=[IntPtr]::Zero
function Get-FG { [Win32]::GetForegroundWindow() }
function Focus-LastWindow {
  try {
    $fg=Get-FG
    if ($fg -ne [IntPtr]::Zero -and $script:form -and $fg -ne $script:form.Handle) { $script:lastHwnd=$fg }
    if ($script:lastHwnd -eq [IntPtr]::Zero) { return $false }
    if ($fg -ne $script:lastHwnd) { [void][Win32]::ShowWindowAsync($script:lastHwnd,5); [void][Win32]::SetForegroundWindow($script:lastHwnd) }
    return $true
  } catch { Write-Log "Focus-LastWindow: $($_.Exception.Message)"; return $false }
}
function Ensure-TargetFast {
  try {
    $fg=Get-FG
    if ($fg -eq [IntPtr]::Zero) { return Focus-LastWindow }
    if ($script:form -and $fg -eq $script:form.Handle) { return Focus-LastWindow }
    $script:lastHwnd=$fg; return $true
  } catch { return $false }
}
