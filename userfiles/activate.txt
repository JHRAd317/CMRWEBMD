Add-Type -AssemblyName System.Drawing

function Play-PayloadWithSleep([string]$payload) {
    if ([string]::IsNullOrWhiteSpace($payload)) { return }
    $parts = [System.Text.RegularExpressions.Regex]::Split($payload, '(\{SLEEP\s+\d+\})') | Where-Object { $_ -ne '' }
    foreach ($p in $parts) {
        if ($p -match '^\{SLEEP\s+(\d+)\}$') {
            $ms = [int]$Matches[1]
            Start-Sleep -Milliseconds $ms
        } else {
            [System.Windows.Forms.SendKeys]::SendWait($p)
        }
    }
}

function Invoke-CustomAction([string]$payload,[string]$mode='sendkeys'){
    if([string]::IsNullOrWhiteSpace($payload)){return}
    switch ($mode) {
      'sendkeys' { Play-PayloadWithSleep $payload }
      default    { [System.Windows.Forms.SendKeys]::SendWait($payload) }
    }
}

$form = New-Object System.Windows.Forms.Form
$form.Text = "CMR Mini Dashboard"
$form.Size = [System.Drawing.Size]::new(400,170)

$btn = New-Object System.Windows.Forms.Button
$btn.Text = "Run Clipboard Payload"
$btn.Size = [System.Drawing.Size]::new(220,42)
$btn.Location = [System.Drawing.Point]::new(20,30)
$btn.Add_Click({
  $payload = [System.Windows.Forms.Clipboard]::GetText()
  if (-not $payload) { [System.Windows.Forms.MessageBox]::Show("Clipboard is empty!","MiniDash","OK","Information") | Out-Null; return }
  Invoke-CustomAction $payload 'sendkeys'
})
$form.Controls.Add($btn)

[void][System.Windows.Forms.Application]::Run($form)
