# --- Enforce STA for Clipboard/WinForms ---
try {
  if ([Threading.Thread]::CurrentThread.ApartmentState -ne 'STA' -and $PSCommandPath -and $env:CMR_NORELAUNCH -ne '1') {
    Start-Process powershell.exe "-STA -WindowStyle Hidden -NoProfile -ExecutionPolicy Bypass -File `"$PSCommandPath`"" | Out-Null
    exit
  }
} catch {}

# --- Main Form Creation ---
$script:form = New-Object System.Windows.Forms.Form
$script:form.Text    = 'CMR Mini Dashboard'
$script:form.ClientSize = [System.Drawing.Size]::new(460,520)
$script:form.StartPosition = 'CenterScreen'
$script:form.BackColor = [System.Drawing.Color]::FromArgb(30,30,30)
$script:form.AutoScaleMode = [System.Windows.Forms.AutoScaleMode]::None
$script:form.TopMost = $true
$script:form.KeyPreview = $true
$script:form.Add_KeyDown({ if($_.KeyCode -eq 'Escape'){ $script:form.Close() } })

# --- Clipboard Bridge Button ---
$btnRun = New-Object System.Windows.Forms.Button
$btnRun.Text = "Run Clipboard Payload"
$btnRun.Size = [System.Drawing.Size]::new(220,42)
$btnRun.Location = [System.Drawing.Point]::new(18,18)
$btnRun.BackColor = [System.Drawing.ColorTranslator]::FromHtml("#1E8449")
$btnRun.ForeColor = [System.Drawing.Color]::White
$btnRun.Font = New-Object System.Drawing.Font("Segoe UI",10,[System.Drawing.FontStyle]::Bold)
$btnRun.Add_Click({
    try {
        $payload = [System.Windows.Forms.Clipboard]::GetText()
        if (-not $payload) { [System.Windows.Forms.MessageBox]::Show("Clipboard is empty! Use the web dashboard to copy a payload first.","CMR MiniDash",'OK','Information') | Out-Null; return }
        Invoke-CustomAction -payload $payload -mode 'sendkeys'
    } catch {
        [System.Windows.Forms.MessageBox]::Show("Error running clipboard payload: $($_.Exception.Message)","CMR MiniDash",'OK','Warning') | Out-Null
    }
})
$script:form.Controls.Add($btnRun)

[void][System.Windows.Forms.Application]::Run($script:form)
